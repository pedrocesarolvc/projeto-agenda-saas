# Stack definida na Etapa 1, secao 1.5: a mesma dos outros dois
# projetos do portfolio, sem tecnologia nova por tecnologia nova.

fastapi            # framework da API (app/main.py)
uvicorn[standard]  # servidor ASGI que sobe o FastAPI
sqlmodel           # ORM (SQLAlchemy + Pydantic) usado em modelos/ e database.py
psycopg2-binary    # driver de conexao com o Postgres
pydantic-settings  # apoio a config.py (leitura de variavel de ambiente)
pytest             # suite de testes, incluindo o teste dos dois tenants (Etapa 2, secao 2.9)
httpx              # TestClient do FastAPI depende dele (usado nos testes)
pyjwt              # emissao/validacao do JWT (Etapa 3, app/auth/jwt.py)
bcrypt             # hash de senha (Etapa 3, app/auth/security.py)
alembic            # migrations do banco (Etapa 4, alembic/)
